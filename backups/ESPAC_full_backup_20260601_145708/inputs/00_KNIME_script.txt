# KNIME Python Script for Updating LCI XML with Flow Variables
# Includes logic for water inputs, pesticide emissions, fertiliser inputs, transformation, direct emissions, and packaging
# Includes computation of direct field emissions from Factors

import knime.scripting.io as knio
import xml.etree.ElementTree as ET
import pandas as pd

# Load input tables
df_xml = knio.input_tables[0].to_pandas()
xml_tree = df_xml.iloc[0, 0]
file_name = df_xml.iloc[0, 1]
emission_factors = knio.input_tables[1].to_pandas()

# Get root element
root = xml_tree.getroot() if isinstance(xml_tree, ET.ElementTree) else xml_tree

# Convert to string for raw replacements
xml_string = ET.tostring(root, encoding='unicode')

# Load flow variables
flow_vars = knio.flow_variables
flow_dict = {k: v for k, v in flow_vars.items()}

# Replace placeholders
# <exchange number="1" name="SimaPro_process_name" location="GLO" unit="kg" meanValue="666" 
# generalComment="Yield per 1 ha, actual area (ha): Area_ha" infrastructureProcess="false">
# Replace placeholders in XML string using values from flow_dict
for key in ['Area_ha']:
    if key in flow_dict:
        xml_string = xml_string.replace(key, str(flow_dict[key]))
# Replace the hardcoded yield and SimaPro name with actual data
Yield_kgha = float(flow_dict.get("Yield_kgha", 0))
xml_string = xml_string.replace('666', str(Yield_kgha))
SPname = flow_dict.get("SimaPro_process_name")
xml_string = xml_string.replace('simapro_process_name', SPname)

# Re-parse XML
root = ET.fromstring(xml_string)

crop_type = str(flow_dict.get("Custom_classification_3", "")).lower() 

# total mass of fertilisers inputs
Urea_kgha = float(flow_dict.get("Urea_kgha", 0))
Limestone_kgha = float(flow_dict.get("Limestone_kgha", 0))
NPK_kgha = float(flow_dict.get("NPK_kgha", 0)) 
NPK_N = float(flow_dict.get("NPK_N_%", 0))  # nutrient contents of NPK mixes
NPK_P = float(flow_dict.get("NPK_P_%", 0))  # nutrient contents of NPK mixes
NPK_K = float(flow_dict.get("NPK_K_%", 0))  # nutrient contents of NPK mixes
MAP_kgha = float(flow_dict.get("MAP_kgha", 0))
UAN_kgha = float(flow_dict.get("UAN_kgha", 0))
AN_kgha = float(flow_dict.get("AN_kgha", 0))
AP_kgha = float(flow_dict.get("AP_kgha", 0))
AS_kgha = float(flow_dict.get("AS_kgha", 0))
CAN_kgha = float(flow_dict.get("CAN_kgha", 0))
Manure_kgha = float(flow_dict.get("Manure_kgha", 0))
Manure_type = str(flow_dict.get("Manure_type", "")) # for selection of manure inputs
Compost_kg_ha = float(flow_dict.get("Compost_kgha", 0))
Digestate_kg_ha = float(flow_dict.get("Digestate_kgha", 0)) 
# Crop residues
Straw_kgha = float(flow_dict.get("Straw_kgha", 0))
Fuel_ha = float(flow_dict.get("Fuel_ha", 0))
export_straw = float(flow_dict.get("export_straw", 0))
crop_res_n_content = {
    "cereal": 0.28/100,     #wheat
    "fruit": 0.20/100,      #orange
    "vegetable": 2.74/100,  #tomate
    "oilseed": 0.80/100,    #sunflower
    "grass": 0.15/100       #fodder
}
N_content_res = crop_res_n_content.get(crop_type, 0.28/100)
N_crop_res = Straw_kgha * (1-export_straw) * N_content_res
#other inputs
Seed_kgha = float(flow_dict.get("Seed_kgha", 0))

################### Direct Field Emission based on IRTA's emission models suit
# Helper function to get value from emission_factors DataFrame
# def get_val(var_name):
#     return emission_factors.loc[emission_factors['Var'] == var_name, 'Val'].values[0]
def get_val(var_name):
    result = emission_factors.loc[emission_factors['Var'] == var_name, 'Val']
    return result.values[0] if not result.empty else 0

# Assign variables
EF_NH3_u = get_val('EF_NH3_u')         # kg NH3-N per kg N applied urea
CF_NH3_u = get_val('EF_NH3_u')         # reduction factor for applied urea, in case of BAT
EF_NH3_m = get_val('EF_NH3_m')         # kg NH3-N per kg N applied cattle manure
CF_NH3_m = get_val('CF_NH3_m')         # reduction factor for applied manure, in case of BAT
FracLEACH = get_val('FracLEACH')       # kg NO3-N per kg N applied IPCC Ch 11 Table 11.3, Tier 1
FracGASF = get_val('FracGASF')         # kg NH3-N per kg inorganic N applied IPCC Ch 11 Table 11.3, Tier 1
FracGASM = get_val('FracGASM')         # kg NH3-N per kg organic N applied IPCC Ch 11 Table 11.3, Tier 1
EF_N2O_d = get_val('EF_N2O_d')         # kg N2O-N per kg N applied
EF_N2O_a = get_val('EF_N2O_a')         # kg N2O-N per kg N-NH3 volatilised
EF_N2O_l = get_val('EF_N2O_l')         # kg N2O-N per kg N-NO3 leached
EF_NOx = FracGASM                      # kg NOx-N per kg N applied
EF_limestone = get_val('EF_limestone') # kg CO2-C per t limestone applied
EF_urea = get_val('EF_urea')           # kg CO2-C per t urea applied
RUSLE_R = get_val('RUSLE_R')           # RUSLE factor
RUSLE_K = get_val('RUSLE_K')           # RUSLE factor
RUSLE_LS = get_val('RUSLE_LS')         # RUSLE factor
RUSLE_C = get_val('RUSLE_C')           # RUSLE factor
RUSLE_P = get_val('RUSLE_P')           # RUSLE factor
Psoil = get_val('Psoil')               # Phosphorus content in soil
e = get_val('e')                       # Enrichment factor
r = get_val('r')                       # Fraction of eroded soil reaching river
t = get_val('t')                       # Land occupation time
kl = get_val('kl')                     # Mean amount of P leached
kr = get_val('kr')                     # Mean amount of P lost to runoff
ks1 = get_val('ks1')                   # correction factor for soil properties (runoff)
ks2 = get_val('ks2')                   # correction factor for soil properties (leaching)
kn = get_val('kn')                     # correction factor for P supply class
kt = get_val('kt')                     # correction factor for topography
s = get_val('s')                       # Slope factor
kd = get_val('kd')                     # Drainage factor
Cd = get_val('mg_Cd')*1e-6
Cu = get_val('mg_Cu')*1e-6
Zn = get_val('mg_Zn')*1e-6
Pb = get_val('mg_Pb')*1e-6
Ni = get_val('mg_Ni')*1e-6
Cr = get_val('mg_Cr')*1e-6
Hg = get_val('mg_Hg')*1e-6

# Emission coefficients
# Selection based on Custom_classification_3 (crop family)
# REFINE TO SELECT SPECIFIC PHASE DEVELOPMENT
c_PrimAir = get_val({'cereal': 'c_PrimAir', 'fruit': 'f_PrimAir'}.get(crop_type, 'v_PrimAir'))
c_PrimSoil = get_val({'cereal': 'c_PrimSoil', 'fruit': 'f_PrimSoil'}.get(crop_type, 'v_PrimSoil'))
c_PrimCrop = get_val({'cereal': 'c_PrimCrop', 'fruit': 'f_PrimCrop'}.get(crop_type, 'v_PrimCrop'))
c_PrimOffField = get_val({'cereal': 'c_PrimOffField', 'fruit': 'f_PrimOffField'}.get(crop_type, 'v_PrimOffField'))
c_PrimWater = get_val({'cereal': 'c_PrimWater', 'fruit': 'f_PrimWater'}.get(crop_type, 'v_PrimWater'))
soil_factor = c_PrimSoil + c_PrimCrop + c_PrimOffField

# Compute total N and P applied
# total nutrient inputs
N_tot = float(flow_dict.get("N_tot", 0))
N_min = float(flow_dict.get("N_min", 0))
N_orga = float(flow_dict.get("N_orga", 0))
P_tot = float(flow_dict.get("P_tot", 0))
P_min = float(flow_dict.get("P_min", 0))
P_orga = float(flow_dict.get("P_orga", 0))
K_tot = float(flow_dict.get("K_tot", 0))
K_min = float(flow_dict.get("K_min", 0))
K_orga = float(flow_dict.get("K_orga", 0))
N_applied = N_tot
P_applied = P_tot

# Compute direct emissions from fertilisers
# Reduction/correction factors must be determined/precalculated in Factors
kgNH3N_ha = ((N_min * EF_NH3_u * CF_NH3_u) + (N_orga * EF_NH3_m * CF_NH3_m)) * 14/17
#kgNOxN_ha = kgN2ON_ha * EF_NOx #IPCC
kgNOxN_ha = (N_applied - kgNH3N_ha) * EF_NOx * 14/46 #EMEP/EEA
kgNO3N_ha = N_applied * FracLEACH #Tier 1, adapted FracLEACH
kgN2ON_ha_direct = (N_applied + N_crop_res) * EF_N2O_d
# version Assumpció et al.
#kgN2ON_ha_indirect = (kgNH3N_ha + kgNOxN_ha + N_min * FracGASF + N_orga * FracGASM) * EF_N2O_a + (kgNO3N_ha * EF_N2O_l)
# version IPCC 2019 (adapted FracGASF and FracGASM)
kgN2ON_ha_indirect = (N_min * FracGASF + N_orga * FracGASM) * EF_N2O_a + (kgNO3N_ha * EF_N2O_l)
kgN2ON_ha = kgN2ON_ha_direct + kgN2ON_ha_indirect
kgCO2urea_ha = Urea_kgha * EF_urea
kgCO2limestone_ha = Limestone_kgha * EF_limestone
soil_erosion_tha = RUSLE_R * RUSLE_K * RUSLE_LS * RUSLE_C * RUSLE_P
fm = P_min
if Manure_type == "Cattle, solid" or "Poultry" in Manure_type:
    fos = P_orga
else:
    fos = 0.0    
if Manure_type == "Cattle, liquid" or Manure_type == "Swine, liquid":
    fol = P_orga
else:
    fol = 0.0
kgP_ha_erosion = soil_erosion_tha * Psoil * e * r 
kgP_ha_runoff = s * kr * ks1 * kn * kt * (1 + 0.2/80 * fm + 0.7/80 * fol + 0.4/80 * fos)
kgP_ha_leaching = kl * ks2 * kn * (1 + 0.2/80 * fol)
kgP_ha_drainage = kgP_ha_leaching * kd
kgP_total = kgP_ha_erosion + kgP_ha_runoff + kgP_ha_leaching + kgP_ha_drainage

########################################################################################

# Update XML
ns = {'eco': 'http://www.EcoInvent.org/EcoSpold01'}
for exchange in root.findall('.//eco:exchange', ns):
    general_comment = str(exchange.get('generalComment', '')).strip()
    name = str(exchange.get('name', '')).strip().lower()

    # Direct match (LCIdata_crops col names vs. ecospold file generalComment)
    if general_comment in flow_dict:
        exchange.set('meanValue', str(flow_dict[general_comment]))
        #print(general_comment + ": " + str(flow_dict[general_comment]))

    # Transformation and occupation logic, "to" and "from"
    if "permanent crop" in name and crop_type == "fruit":
        exchange.set('meanValue', '10000')
        continue
    elif "annual crop" in name and crop_type in ["cereal", "vegetable"]:
        exchange.set('meanValue', '10000')
        continue
    elif "grassland" in name and crop_type == "grass":
        exchange.set('meanValue', '10000')
        continue

    # Packaging logic
    # Packaging_type1 = plastics, Packaging_type2 = other packaging materials
    Packaging_type1 = str(flow_dict.get("Packaging_type1", ""))
    Packaging_type1_kgha = float(flow_dict.get("Packaging_type1_kgha", 0))
    Packaging_type2 = str(flow_dict.get("Packaging_type2", ""))
    Packaging_type2_kgha = float(flow_dict.get("Packaging_type2_kgha", 0))
    PP = (1 if Packaging_type1 == "PP" else 0) * Packaging_type1_kgha
    PE = (1 if Packaging_type1 == "PE" else 0) * Packaging_type1_kgha
    PET = (1 if Packaging_type1 == "PET" else 0) * Packaging_type1_kgha
    LDPE = (1 if Packaging_type1 == "LDPE" else 0) * Packaging_type1_kgha
    Kraft = (1 if Packaging_type2 == "Kraft" else 0) * Packaging_type2_kgha
    Cardboard = (1 if Packaging_type2 == "Cardboard" else 0) * Packaging_type2_kgha
    packaging_map = {
        "PP": PP,
        "PE": PE,
        "PET": PET,
        "LDPE": LDPE,
        "Kraft": Kraft,
        "Cardboard": Cardboard
    }
    for key, value in packaging_map.items():
        if key in general_comment:
            exchange.set('meanValue', str(value))
            continue

    # Water logic
    water_source = str(flow_dict.get("Water_source", "")).lower()
    water_m3 = float(flow_dict.get("Irrig_m3", 0))
    water_well = 1 if water_source == "well" else 0
    water_river = 1 if water_source == "river" else 0
    water_lake = 1 if water_source == "lake" else 0
    # Water inputs
    if "water, well" in name:
        exchange.set('meanValue', str(water_m3 * water_well))
        continue
    elif "water, lake" in name:
        exchange.set('meanValue', str(water_m3 * water_lake))
        continue
    elif "water, river" in name:
        exchange.set('meanValue', str(water_m3 * water_river))
        continue

    # Fuel logic
    fuel_type = str(flow_dict.get("Fuel_type", ""))
    fuel_unit = str(flow_dict.get("Fuel_unit", ""))
    # Fuel inputs and combustion
    if "Fuel" in general_comment and "Diesel" in fuel_type and fuel_unit == "L":
        exchange.set('meanValue', str(Fuel_ha * 38)) 
        # LHV Diesel = 38 MJ/l
        continue
    elif "Fuel" in general_comment and "Gasoline" in fuel_type and fuel_unit == "L":
        exchange.set('meanValue', str(Fuel_ha * 38)) 
        # LHV Gasoline = 33.6 MJ/l
        continue        

    # Fertiliser inputs 
    #!!!!!!!!!! COMPLETAR PARA TODOS LOS TIPOS DE Manure_type Y PARA FERTILISERS NOT AS NUTRIENTS!!!!!!!!!!!!!!!!!!!!!!!!!!
    Fert_as_nutrients = str(flow_dict.get("Fert_as_nutrients", "")) # for selection of fertiliser modelling approach: as nutrients or as products
    if Fert_as_nutrients == "yes":
        if "inorganic nitrogen fertiliser" in name:
            exchange.set('meanValue', str(N_min))
        elif "inorganic phosphorus fertiliser" in name:
            exchange.set('meanValue', str(P_min))
        elif "inorganic potassium fertiliser" in name:
            exchange.set('meanValue', str(K_min))
        elif "manure, solid, cattle" in name and "nitrogen" in name and Manure_type == "Cattle, solid":
            exchange.set('meanValue', str(N_orga))
        elif "manure, solid, cattle" in name and "phosphorus" in name and Manure_type == "Cattle, solid":
            exchange.set('meanValue', str(P_orga))
        elif "manure, solid, cattle" in name and "potassium" in name and Manure_type == "Cattle, solid":
            exchange.set('meanValue', str(K_orga))
        elif "manure, liquid, cattle" in name and "nitrogen" in name and Manure_type == "Cattle, liquid":
            exchange.set('meanValue', str(N_orga))
        elif "manure, liquid, cattle" in name and "phosphorus" in name and Manure_type == "Cattle, liquid":
            exchange.set('meanValue', str(P_orga))
        elif "manure, liquid, cattle" in name and "potassium" in name and Manure_type == "Cattle, liquid":
            exchange.set('meanValue', str(K_orga))
        elif "manure, liquid, swine" in name and "nitrogen" in name and Manure_type == "Swine, liquid":
            exchange.set('meanValue', str(N_orga))
        elif "manure, liquid, swine" in name and "phosphorus" in name and Manure_type == "Swine, liquid":
            exchange.set('meanValue', str(P_orga))
        elif "manure, liquid, swine" in name and "potassium" in name and Manure_type == "Swine, liquid":
            exchange.set('meanValue', str(K_orga))
        elif "poultry manure" in name and "nitrogen" in name and "Poultry" in Manure_type:
            exchange.set('meanValue', str(N_orga))
        elif "poultry manure" in name and "phosphorus" in name and "Poultry" in Manure_type:
            exchange.set('meanValue', str(P_orga))
        elif "poultry manure" in name and "potassium" in name and "Poultry" in Manure_type:
            exchange.set('meanValue', str(K_orga))
    else :
        if "Cattle, solid" in general_comment and "Cattle, solid" in Manure_type:
            exchange.set('meanValue', str(Manure_kgha))

    # Seed
    # Prefer exchange-name-specific routing first (wheat/carrot/rape), then crop_type fallback.
    seed_bucket = None
    name_l = name.lower()
    gc_l = general_comment.lower()
    if "wheat seed" in name_l:
        seed_bucket = "cereal"
    elif "carrot seed" in name_l:
        seed_bucket = "vegetable"
    elif "rape seed" in name_l or "rapeseed" in name_l:
        seed_bucket = "oilseed"
    else:
        if "seed, cereal" in gc_l:
            seed_bucket = "cereal"
        elif "seed, vegetable" in gc_l:
            seed_bucket = "vegetable"
        elif "seed, oilseed" in gc_l:
            seed_bucket = "oilseed"

    if seed_bucket is not None:
        if crop_type in ["cereal", "vegetable", "oilseed"]:
            target_seed_bucket = crop_type
        else:
            target_seed_bucket = "cereal"
        exchange.set('meanValue', str(Seed_kgha if seed_bucket == target_seed_bucket else 0.0))
        continue

    # Direct emissions
    if "NH3" in general_comment:
        exchange.set('meanValue', str(kgNH3N_ha))
    elif "N2O" in general_comment:
        exchange.set('meanValue', str(kgN2ON_ha))
    elif "NOx" in general_comment:
        exchange.set('meanValue', str(kgNOxN_ha))
    elif "CO2" in general_comment:
        exchange.set('meanValue', str(kgCO2urea_ha+kgCO2limestone_ha))
    elif "NO3" in general_comment:
        exchange.set('meanValue', str(kgNO3N_ha))
    elif name.startswith("phosphorus") or "phosphorus" in name and general_comment == "P":
        exchange.set('meanValue', str(kgP_total))

    # Trace elements from fertilisers
    metals_map = {
        "Cd": Cd,
        "Cu": Cu,
        "Zn": Zn,
        "Pb": Pb,
        "Ni": Ni,
        "Cr": Cr,
        "Hg": Hg
    }
    for key, value in metals_map.items():
        if key in general_comment:
            exchange.set('meanValue', str(value))
            continue

    # Pesticide names and doses
    pesticides = []
    for i in range(1, 10):
        pest_name = str(flow_dict.get(f"Molecule_name_{i}", "")).strip().lower()
        if pest_name != "missing":
            dose = float(flow_dict.get(f"Dose_kgha_{i}", 0))
            pesticides.append((pest_name, dose))
    # Pesticide emissions
    # <exchange number="55" category="air" subCategory="unspecified" name="Azoxystrobin" unit="kg" meanValue="0" uncertaintyType="0" generalComment="c_PrimAir">
    else:
        for pest_name, dose in pesticides:
            if pest_name == name:
                print(name)
                if general_comment == "c_PrimAir":
                    exchange.set('meanValue', str(dose * c_PrimAir/100))
                elif general_comment == "soil_factor":
                    exchange.set('meanValue', str(dose * soil_factor/100))
                elif general_comment == "c_PrimWater":
                    exchange.set('meanValue', str(dose * c_PrimWater/100))

    # Transport types and tkm
    transport = []
    for i in range(1, 10):
        transport_type = str(flow_dict.get(f"Transport_type_{i}", "")).strip()
        tkm = float(flow_dict.get(f"Transport_tkm_{i}", 0))
        if transport_type != "":
            transport.append((transport_type, tkm))
    totals = {}
    for name, value in transport:
        totals[name] = totals.get(name, 0) + value
    # Transport
    # <exchange number="40" name="transport, tractor and trailer, agricultural {CH}| market for transport, tractor and trailer, agricultural | Cut-off, U" location="GLO" unit="tkm" meanValue="0" uncertaintyType="0" generalComment="Tractor"
    for transport_type, tkm in totals.items():
        if general_comment == "Truck" and transport_type == "Truck":
            exchange.set('meanValue', str(tkm))
        elif general_comment == "Tractor" and transport_type == "Tractor":
            exchange.set('meanValue', str(tkm))

# Output
#updated_xml_str = ET.tostring(root, encoding='unicode')
updated_xml_bytes = ET.tostring(root, encoding='utf-8')
updated_xml_str = updated_xml_bytes.decode('utf-8')
updated_xml_str = updated_xml_str.replace("ns0:","")
updated_xml_str = updated_xml_str.replace('<ecoSpold xmlns:ns0="http://www.EcoInvent.org/EcoSpold01" xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xsi:schemaLocation="http://www.EcoInvent.org/EcoSpold01 http://www.EcoInvent.org/EcoSpold01..\..\EcoSpold01Dataset.xsd">','<ecoSpold xmlns:xsi="http://www.w3.org/2001/XMLSchema-instance" xmlns="http://www.EcoInvent.org/EcoSpold01" xsi:schemaLocation="http://www.EcoInvent.org/EcoSpold01 http://www.EcoInvent.org/EcoSpold01..\..\EcoSpold01Dataset.xsd https://github.com/brightway-lca/pyecospold/tree/main/pyecospold/schemas/v1">')
original = updated_xml_str
insert_after = "</representativeness>"
insertion = "\n<validation>\n<reviewStatus>Final</reviewStatus>\n<reviewer>Angel AVADI</reviewer>\n<reviewDate>2025-11-06</reviewDate>\n<proofReadingValidator>\n<name>Angel AVADI</name>\n<email>angel.avadi@irta.cat</email>\n<companyCode>1234</companyCode>\n<proofReadingDate>2025-11-06</proofReadingDate>\n</proofReadingValidator>\n</validation>"
index = original.find(insert_after)
if index != -1:
    index += len(insert_after)
    updated_xml_str = original[:index] + insertion + original[index:]
#updated_xml_str = updated_xml_str.replace('<validation>', '<validation proofReadingValidator="true">')

output_df = pd.DataFrame({
    df_xml.columns[0]: [updated_xml_str],
    'file_name': [file_name]
})

# Output comparison table
# INTEGRATE TRACE ELEMENTS
comparison_table = pd.DataFrame({
    "kgNO3N_ha_original": [float(flow_dict.get("kgNO3N_ha", 0))],
    "kgNO3N_ha_new": [kgNO3N_ha],
    "kgNH3N_ha_original": [float(flow_dict.get("kgNH3N_ha", 0))],
    "kgNH3N_ha_new": [kgNH3N_ha],
    "kgN2ON_ha_original": [float(flow_dict.get("kgN2ON_ha", 0))],
    "kgN2ON_ha_new": [kgN2ON_ha],
    "kgNOxN_ha_original": [float(flow_dict.get("kgNOxN_ha", 0))],
    "kgNOxN_ha_new": [kgNOxN_ha],
    "kgCO2urea_ha_original": [float(flow_dict.get("kgCO2urea_ha", 0))],
    "kgCO2urea_ha_new": [kgCO2urea_ha],
    "kgP_ha_erosion_original": [float(flow_dict.get("kgP_ha_erosion", 0))],
    "kgP_ha_erosion_new": [kgP_ha_erosion],
    "kgP_ha_runoff_original": [float(flow_dict.get("kgP_ha_runoff", 0))],
    "kgP_ha_runoff_new": [kgP_ha_runoff]
})

knio.output_tables[0] = knio.Table.from_pandas(output_df)
knio.output_tables[1] = knio.Table.from_pandas(comparison_table)
